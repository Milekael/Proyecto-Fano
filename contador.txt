<body>
    <div class="contador" id="contador">
        <p id="recuento">Cargando...</p>
    </div>

    <script>
        // Fecha de destino (24 de junio de 2025, 3 PM)
        var fechaDestino = new Date("Jun 24, 2025 15:00:00").getTime();

        // Actualizar el contador cada segundo
        var x = setInterval(function() {

            // Obtener la fecha y hora actual
            var ahora = new Date().getTime();

            // Calcular la distancia entre la fecha actual y la fecha destino
            var distancia = fechaDestino - ahora;

            // Calcular días, horas, minutos y segundos
            var dias = Math.floor(distancia / (1000 * 60 * 60 * 24));
            var horas = Math.floor((distancia % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            var minutos = Math.floor((distancia % (1000 * 60 * 60)) / (1000 * 60));
            var segundos = Math.floor((distancia % (1000 * 60)) / 1000);

            // Mostrar el resultado en el elemento con id "recuento"
            document.getElementById("recuento").innerHTML = dias + "d " + horas + "h "
            + minutos + "m " + segundos + "s ";

            // Si la cuenta atrás ha terminado, escribir un mensaje
            if (distancia < 0) {
                clearInterval(x);
                document.getElementById("recuento").innerHTML = "¡El evento ha llegado!";
            }
        }, 1000);
    </script>
